6   package opetusohjelma.kayttoliittyma.controller;
7   
8   import java.awt.event.ActionEvent;
9   import java.awt.event.ActionListener;
10  import java.util.ArrayList;
11  import javax.swing.JLabel;
12  
13  /**
14   * Tämän luokan avulla kuunnellaan "New assignment-näppäintä".
15   *
16   * @author Iisa
17   */
18  public class NewAssignmentNapinKuuntelija implements ActionListener {
19  
20      private JLabel tehtavakentta;
21      private ArrayList<String> funktioJaVastaus;
22      private ShowSolutionNapinKuuntelija skuuntelija;
23  
24      /**
25       * Tämä on konstruktori, jossa asetetaan private-muuttujat headerLabel,
26       * funktiojaVastaus ja skuuntelija ympäristöstä, josta
27       * ValinnanNewAssignmentKuuntelijaa kutsutaan.
28       *
29       * @param tehtavakentta JLabel
30       * @param funktioJaVastaus ArrayList
31       * @param skuuntelija ShowSolutionKuuntelija
32       */
33      public NewAssignmentNapinKuuntelija(JLabel tehtavakentta, ArrayList<String> funktioJaVastaus, ShowSolutionNapinKuuntelija skuuntelija) {
34          this.tehtavakentta = tehtavakentta;
35          this.funktioJaVastaus = funktioJaVastaus;
36          this.skuuntelija = skuuntelija;
37      }
38  
39      /**
40       * Tämän metodin avulla kuunnellaan "New assignment"-näppäintä. Kuuntelun
41       * tuloksena tietokone arpoo uuden tehtävän, joka asetetaan taas
42       * headerLabel:iin. Uuden funktio ja sen vastaukset lisätään
43       * ValinnanShowSolutionKuuntelijaan.
44       *
45       * @param e
46       */
47      @Override
48      public void actionPerformed(ActionEvent e) {
49          Arpoja arpoja = new Arpoja();
50          String toiminto = arpoja.arvoToiminta();
51          this.funktioJaVastaus = arpoja.arvoFunktioJaVastaukset(arpoja.arvoFunktio());
52          String funktio = funktioJaVastaus.get(0);
53          String tehtava = arpoja.tulostaTehtava(toiminto, funktio);
54          tehtavakentta.setText(tehtava);
55  
56          skuuntelija.setFunktioJaVastaus(funktioJaVastaus);
57          skuuntelija.setToiminto(toiminto);
58      }
59  
60  }
