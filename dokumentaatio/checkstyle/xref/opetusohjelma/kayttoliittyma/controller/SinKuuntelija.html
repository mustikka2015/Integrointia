6   package opetusohjelma.kayttoliittyma.controller;

8   import java.awt.event.ActionEvent;
9   import java.awt.event.ActionListener;
10  import javax.swing.SwingUtilities;
11  import opetusohjelma.kayttoliittyma.view.SinGUI;
12  
13  /**
14   * Tämän luokan avulla kuunnellaan "Sine"-näppäintä.
15   *
16   * @author Iisa
17   */
18  public class SinKuuntelija implements ActionListener {
19      
20       /**
21       * Tämän metodin avulla kuunnellaan Sine-näppäintä. Kuuntelun tuloksena
22       * tietokone näyttää uuden käyttöliittymänäkymän, johon opiskelija syöttää 
23       * funktion kertoimen ja sen sisäfunktion kertoimen ja valitsee, integroidaanko vai 
24       * derivoidaanko funktio.
25       */
26      @Override
27      public void actionPerformed(ActionEvent e) {
28          SinGUI sinGUI = new SinGUI();
29          SwingUtilities.invokeLater((Runnable) sinGUI);
30      }
31  
32  }
