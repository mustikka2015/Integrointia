6   package opetusohjelma.kayttoliittyma.controller;

8   import java.util.ArrayList;
9   import opetusohjelma.laskutoimituksia.Polynomi;
10  import opetusohjelma.laskutoimituksia.SinCos;
11  
12  /**
13   * Luokka toteuttaa tehtävän arpomiseen liittyviä toiminnallisuuksia.
14   *
15   * @author Iisa
16   */
17  public class TehtavanArpoja {
18  
19      private ArrayList<String> funktioJaVast;
20  
21      /**
22       * Konstruktori luokalle TehtavanArpoja.
23       */
24      public TehtavanArpoja() {
25  
26      }
27  
28      /**
29       * Metodi arpoo funktion. Se palauttaa ArrayList-olion, joka sisältää itse
30       * funktion String-muodossa, funktion derivoituna String-muodossa sekä
31       * funktion integroituna String-muodossa.
32       *
33       * @return ArrayList<String>, joka sisältää sekä funktion että sen
34       * derivaatan ja integraalin.
35       */
36      public ArrayList<String> arvoFunktio() {
37          Arpoja arpoja = new Arpoja();
38          ArrayList<String> funktioJaVast = new ArrayList<String>();
39          String funktio = arpoja.arvoFunktio();
40          if (funktio.equals("sin") || funktio.equals("cos")) {
41              funktioJaVast = sinCosFunktionArpominen(funktioJaVast, funktio, arpoja);
42  
43          } else {
44              funktioJaVast = polynomifunktionArpominen(funktioJaVast, arpoja);
45          }
46  
47          return funktioJaVast;
48      }
49  
50      /**
51       * Metodi arpoo sini- tai kosinifunktion kertoimet. Se palauttaa
52       * ArrayList-olion, joka sisältää itse funktion String-muodossa, funktion
53       * derivoituna String-muodossa sekä funktion integroituna String-muodossa.
54       *
55       * @param funktioJaVast ArrayList<String>-muodossa
56       * @param funktio String-muodossa
57       * @param arpoja Arpoja-muodossa
58       *
59       * @return ArrayList<String>, joka sisältää sekä funktion että sen
60       * derivaatan ja integraalin.
61       */
62      public ArrayList<String> sinCosFunktionArpominen(ArrayList<String> funktioJaVast, String funktio, Arpoja arpoja) {
63          double kerroin1 = arpoja.arvoKerroin();
64          double kerroin2 = arpoja.arvoKerroin();
65          SinCos sincos = new SinCos(kerroin1, kerroin2, funktio);
66          funktio = sincos.toString();
67          funktioJaVast.add(funktio);
68          SinCos derivoitu = new SinCos(sincos.getKerroin(), sincos.getSisafunktionKerroin(), sincos.getFunktio());
69          SinCos integroitu = new SinCos(sincos.getKerroin(), sincos.getSisafunktionKerroin(), sincos.getFunktio());
70          derivoitu.derivoi();
71          integroitu.integroi();
72          funktioJaVast.add(derivoitu.toString());
73          funktioJaVast.add(integroitu.toString());
74          return funktioJaVast;
75      }
76  
77      /**
78       * Metodi arpoo polynomifunktion kertoimen ja eksponentin. Se palauttaa
79       * ArrayList-olion, joka sisältää itse funktion String-muodossa, funktion
80       * derivoituna String-muodossa sekä funktion integroituna String-muodossa.
81       *
82       * @param funktioJaVast ArrayList<String>-muodossa
83       * @param arpoja Arpoja-muodossa
84       *
85       * @return ArrayList<String>, joka sisältää sekä funktion että sen
86       * derivaatan ja integraalin.
87       */
88      public ArrayList<String> polynomifunktionArpominen(ArrayList<String> funktioJaVast, Arpoja arpoja) {
89          double kerroin = arpoja.arvoKerroin();
90          int eks = arpoja.arvoPolynominEksponentti();
91          Polynomi polynomi = new Polynomi(eks, kerroin);
92          String funktio = polynomi.toString();
93          funktioJaVast.add(funktio);
94          Polynomi derivoitu = new Polynomi(polynomi.getEksponentti(), polynomi.getKerroin());
95          Polynomi integroitu = new Polynomi(polynomi.getEksponentti(), polynomi.getKerroin());
96          derivoitu.derivoi();
97          integroitu.integroi();
98          funktioJaVast.add(derivoitu.toString());
99          funktioJaVast.add(integroitu.toString());
100         return funktioJaVast;
101     }
102 
103     /**
104      * Metodi arpoo, integroidaanko vai derivoidaanko funktio.
105      *
106      * @return String: "Integrate" tai "Derivate"
107      */
108     public String arvoToiminto() {
109         Arpoja arpoja = new Arpoja();
110         String toiminto = arpoja.arvoToiminto();
111         return toiminto;
112     }
113 
114     /**
115      * Metodi kertoo tehtävänannon String-muodossa, kun sille on syötetty jo
116      * aikaisemmin arvotut toiminto ja funktio.
117      *
118      * @param toiminto String-muodossa
119      * @param funktio String-muodossa
120      *
121      * @return String: tehtävänanto
122      */
123     public String arvoTehtava(String toiminto, String funktio) {
124 
125         String tehtava = toiminto;
126         tehtava = tehtava + " function y = " + funktio;
127         tehtava = tehtava + ". Click to check.";
128         return tehtava;
129     }
130 }
