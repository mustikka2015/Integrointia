6   package opetusohjelma.kayttoliittyma.controller;
7   
8   import java.util.ArrayList;
9   import java.util.Random;
10  
11  /**
12   * Luokka tarjoaa funktion, eksponentin ja kertoimen arpomiseen tarvittavia
13   * metodeja.
14   *
15   * @author Iisa
16   */
17  public class Arpoja {
18  
19      private Random arpoja;
20  
21      /**
22       * Metodi alustaa Random-tyyppisen Arpoja-olion. 
23       */
24      public Arpoja() {
25          this.arpoja = new Random();
26      }
27  
28      /**
29       * Metodi arpoo funktion, jonka opiskelijan tulee integroida tai derivoida.
30       * Funktiovaihtoehdot ovat polynomi-, sini- ja kosinifunktio.
31       *
32       * @return String: "polynom", "sin" tai "cos".
33       */
34      public String arvoFunktio() {
35          ArrayList<String> funktiot = new ArrayList<String>();
36          funktiot.add("polynom");
37          funktiot.add("sin");
38          funktiot.add("cos");
39  
40          int arvottu = this.arpoja.nextInt(funktiot.size());
41  
42          String funktio = funktiot.get(arvottu);
43  
44          return funktio;
45      }
46  
47      /**
48       * Metodi arpoo polynomifunktion eksponentin. Eksponentti voi saada arvoja
49       * -10:stä 10:een.
50       *
51       * @return arvottu eksponentti int-muodossa.
52       */
53      public int arvoPolynominEksponentti() {
54  
55          int arvottu = this.arpoja.nextInt(21);
56          arvottu = arvottu - 10;
57  
58          return arvottu;
59      }
60  
61      /**
62       * Metodi arpoo double-tyyppisen kertoimen. Kerroin voi saada arvoja
63       * -10.0:stä 10.0:een.
64       *
65       * @return arvottu kerroin double-muodossa.
66       */
67      public double arvoKerroin() {
68  
69          double arvottu = (double) (this.arpoja.nextInt(201) - 100) / 10;
70  
71          return arvottu;
72      }
73  
74      /**
75       * Metodi arpoo, integroidaanko vai derivoidaanko funktio.
76       *
77       * @return String: "Integrate" tai "Derivate".
78       */
79      public String arvoToiminto() {
80  
81          ArrayList<String> vaihtoehdot = new ArrayList<String>();
82  
83          vaihtoehdot.add("Integrate");
84          vaihtoehdot.add("Derivate");
85  
86          int arvottu = this.arpoja.nextInt(vaihtoehdot.size());
87  
88          String toiminta = vaihtoehdot.get(arvottu);
89  
90          return toiminta;
91      }
92  
93  }
