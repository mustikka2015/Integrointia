1   package opetusohjelma.kayttoliittyma;
2   
3   import java.util.ArrayList;
4   import java.util.Random;
5   
6   /**
7    * Tämän luokan avulla luodaan olioita, jotka arpovat sekä itse funktion, että
8    * sen kertoimet.
9    */
10  public class Arpoja {
11  
12      private Random arpoja;
13  
14      /**
15       * Luokka tarjoaa funktion, eksponentin ja kertoimen arpomiseen tarvittavia
16       * metodeja.
17       */
18      public Arpoja() {
19          this.arpoja = new Random();
20      }
21  
22      /**
23       * Metodi arpoo funktion, jonka opiskelijan tulee integroida tai derivoida.
24       * Funktiovaihtoehdot ovat polynomi-, sini- ja kosinifunktio.
25       *
26       * @return arvottu funktio String-muodossa
27       */
28      public String arvoFunktio() {
29          ArrayList<String> funktiot = new ArrayList<String>();
30          funktiot.add("polynom");
31          funktiot.add("sin");
32          funktiot.add("cos");
33  
34          int arvottu = this.arpoja.nextInt(funktiot.size());
35  
36          String funktio = funktiot.get(arvottu);
37  
38          return funktio;
39      }
40  
41      /**
42       * Metodi arpoo polynomifunktion eksponentin. Eksponentti voi saada arvoja
43       * -10:stä 10:een.
44       *
45       * @return arvottu eksponentti int-muodossa.
46       */
47      public int arvoPolynominEksponentti() {
48  
49          int arvottu = this.arpoja.nextInt(21);
50          arvottu = arvottu - 10;
51  
52          return arvottu;
53      }
54  
55      /**
56       * Metodi arpoo double-tyyppisen kertoimen. Kerroin voi saada arvoja
57       * -10.0:stä 10.0:een.
58       *
59       * @return arvottu kerroin double-muodossa.
60       */
61      public double arvoKerroin() {
62  
63          double arvottu = (double) (this.arpoja.nextInt(201) - 100) / 10;
64  
65          return arvottu;
66      }
67  
68      /**
69       * Metodi arpoo, integroidaanko vai derivoidaanko funktio.
70       *
71       * @return String: "Integroi" tai "Derivoi"
72       */
73      public String arvoToiminto() {
74  
75          ArrayList<String> vaihtoehdot = new ArrayList<String>();
76  
77          vaihtoehdot.add("Integroi");
78          vaihtoehdot.add("Derivoi");
79  
80          int arvottu = this.arpoja.nextInt(vaihtoehdot.size());
81  
82          String toiminta = vaihtoehdot.get(arvottu);
83  
84          return toiminta;
85      }
86  
87  }
