6   package opetusohjelma.kayttoliittyma.view;
7   
8   import java.awt.*;
9   import java.awt.event.*;
10  import javax.swing.*;
11  
12  import java.awt.CardLayout;
13  import opetusohjelma.kayttoliittyma.controller.ChangeKuuntelija;
14  import opetusohjelma.kayttoliittyma.controller.CosKuuntelija;
15  import opetusohjelma.kayttoliittyma.controller.PolynomKuuntelija;
16  import opetusohjelma.kayttoliittyma.controller.StartKuuntelija;
17  import opetusohjelma.kayttoliittyma.controller.SinKuuntelija;
18  
19  /**
20   * Käyttöliittymän aloitussivusto.
21   */
22  public class AloitusGUI {
23  
24      private JFrame mainFrame;
25      private JLabel headerLabel;
26      private JLabel statusLabel;
27      private JPanel controlPanel;
28  
29      /**
30       * Konstruktori AloitusGUI:lle.
31       */
32      public AloitusGUI() {
33          prepareGUI();
34      }
35  
36      /**
37       * Metodi luo JFrame-olion ja alustaa sen halutunlaiseksi.
38       */
39      public void prepareGUI() {
40          mainFrame = new JFrame("Integration and differentiation");
41          mainFrame.setSize(400, 500);
42          mainFrame.setLayout(new GridLayout(3, 1));
43  //        Container c = mainFrame.getContentPane();
44  //        c.setBackground(Color.blue);
45  
46          headerLabel = new JLabel("", JLabel.CENTER);
47          statusLabel = new JLabel("", JLabel.CENTER);
48          statusLabel.setSize(350, 100);
49  
50          mainFrame.addWindowListener(new WindowAdapter() {
51              public void windowClosing(WindowEvent windowEvent) {
52                  System.exit(0);
53              }
54          });
55          controlPanel = new JPanel();
56          controlPanel.setLayout(new FlowLayout());
57  
58          mainFrame.add(headerLabel);
59          mainFrame.add(controlPanel);
60          mainFrame.add(statusLabel);
61          mainFrame.setVisible(true);
62      }
63  
64      /**
65       * Metodi luo CardLayout-näkymän ja sen paneelit painikkeineen. Tämän
66       * refaktorointi on vielä kesken.
67       */
68      public void showCardLayout() {
69          headerLabel.setText("Choose a function or draw lots to decide it");
70  
71          JPanel panel = new JPanel();
72  
73          panel.setSize(300, 300);
74  
75          CardLayout layout = new CardLayout();
76          layout.setHgap(10);
77          layout.setVgap(10);
78          panel.setLayout(layout);
79  
80          JPanel buttonPanel = new JPanel(new FlowLayout());
81  
82          lisaaNappi(buttonPanel, new JButton("Polynom"), new PolynomKuuntelija());
83  
84          lisaaNappi(buttonPanel, new JButton("Sine"), new SinKuuntelija());
85  
86          lisaaNappi(buttonPanel, new JButton("Cosine"), new CosKuuntelija());
87  
88          JPanel selectPanel = new JPanel(new FlowLayout());
89  
90          JButton start = new JButton("Start!");
91          start.setBackground(Color.GREEN);
92  
93          lisaaNappi(selectPanel, start, new StartKuuntelija());
94  
95          panel.add("You choose", buttonPanel);
96          panel.add("Computer chooses", selectPanel);
97          DefaultComboBoxModel panelName = new DefaultComboBoxModel();
98  
99          panelName.addElement("You choose");
100         panelName.addElement("Computer chooses");
101         JComboBox listCombo = new JComboBox(panelName);
102 
103         listCombo.setSelectedIndex(0);
104 
105         JScrollPane listComboScrollPane = new JScrollPane(listCombo);
106 
107         controlPanel.add(listComboScrollPane);
108 
109         JButton change = new JButton("Change");
110 
111         lisaaNappi(controlPanel, change, new ChangeKuuntelija(listCombo, panel, statusLabel));
112 
113         controlPanel.add(panel);
114 
115         mainFrame.setVisible(true);
116     }
117 
118     /**
119      * Metodi lisää painikkeen lisäten siihen tapahtumankuuntelijan.
120      *
121      * @param panel JPanel
122      * @param button JButton
123      * @param listener ActionListener
124      */
125     public void lisaaNappi(JPanel panel, JButton button, ActionListener listener) {
126 
127         button.addActionListener(listener);
128         panel.add(button);
129 
130     }
131 
132 }
